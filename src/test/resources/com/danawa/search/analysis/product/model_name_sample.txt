################################################################################
# 특수문자
# 특수문자는 색인 검색시 삭제 
# "-", "/" 만 인식 +는 기본적으로 미포함
abcde1234#	abcde1234 <MODEL_NAME>	abcde <ALPHA>	1234 <NUMBER>
a-1	a-1 <MODEL_NAME>
1-a	1-a <MODEL_NAME>
a#b	a <ALPHA>	b <ALPHA>
a+b	a <ALPHA>	b <ALPHA>
a/b	a/b <MODEL_NAME>
#a/	a <ALPHA>
1-	1 <NUMBER>
c+	c <ALPHA>
# + 는 기본적으로 미포함되지만 pa++ 는 사전에 존재하므로 HANGUL 로 뽑힘
# pa++	pa <ALPHA>
pa++	pa++ <HANGUL>

#1-2. 삭제된 특수문자로 구분되어 형태소 분리가 된다.

abc#abc	abc <ALPHA>	abc <ALPHA>


################################################################################
